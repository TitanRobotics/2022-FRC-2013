001package org.usfirst.frc2022.commands;
002
003import edu.wpi.first.wpilibj.command.Command;
004import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
005import org.usfirst.frc2022.OI;
006import org.usfirst.frc2022.RobotMap;
007import org.usfirst.frc2022.subsystems.CameraServos;
008import org.usfirst.frc2022.subsystems.ExampleSubsystem;
009import org.usfirst.frc2022.subsystems.Robocam;
010import org.usfirst.frc2022.subsystems.PWM_Generic;
011
012/**
013 * The base for all commands. All atomic commands should subclass CommandBase.
014 * CommandBase stores creates and stores each control system. To access a
015 * subsystem elsewhere in your code in your code use
016 * CommandBase.exampleSubsystem
017 * 
018 * @author Author
019 */
020public abstract class CommandBase extends Command {
021    
022        public static OI oi;
023        // Create a single static instance of all of your subsystems
024        //public static ExampleSubsystem exampleSubsystem = new ExampleSubsystem();
025        public static Robocam cam = new Robocam("10.20.22.11");
026        public static CameraServos camServos = new CameraServos();
027        public static PWM_Generic pwmGeneric = new PWM_Generic(RobotMap.portsJaguar);
028
029        public static void init() {
030                // This MUST be here. If the OI creates Commands (which it very likely
031                // will), constructing it during the construction of CommandBase (from
032                // which commands extend), subsystems are not guaranteed to be
033                // yet. Thus, their requires() statements may grab null pointers. Bad
034                // news. Don't move it.
035                oi = new OI();
036
037                // Show what command your subsystem is running on the SmartDashboard
038                //SmartDashboard.putData(exampleSubsystem);
039                SmartDashboard.putData(cam);
040                SmartDashboard.putData(camServos);
041                SmartDashboard.putData(pwmGeneric);
042        }
043
044        public CommandBase(String name) {
045                super(name);
046        }
047
048        public CommandBase() {
049                super();
050        }
051}

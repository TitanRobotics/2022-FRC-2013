001package org.usfirst.frc2022.commands;
002
003import edu.wpi.first.wpilibj.command.Command;
004import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
005import org.usfirst.frc2022.OI;
006import org.usfirst.frc2022.RobotMap;
007import org.usfirst.frc2022.subsystems.CameraServos;
008import org.usfirst.frc2022.subsystems.PWM_Generic;
009import org.usfirst.frc2022.subsystems.Robocam;
010
011/**
012 * The base for all commands. All atomic commands should subclass CommandBase.
013 * CommandBase stores creates and stores each control system. To access a
014 * subsystem elsewhere in your code in your code use
015 * CommandBase.exampleSubsystem
016 * 
017 * @author Titan Robotics (2022)
018 */
019public abstract class CommandBase extends Command {
020    
021        public static OI oi;
022        // Create a single static instance of all of your subsystems
023        //public static ExampleSubsystem exampleSubsystem = new ExampleSubsystem();
024        public static Robocam cam = new Robocam("10.20.22.11");
025        public static CameraServos camServos = new CameraServos();
026        public static PWM_Generic pwmDriveBase = new PWM_Generic(RobotMap.portsJaguar);
027        public static void init() {
028                // This MUST be here. If the OI creates Commands (which it very likely
029                // will), constructing it during the construction of CommandBase (from
030                // which commands extend), subsystems are not guaranteed to be
031                // yet. Thus, their requires() statements may grab null pointers. Bad
032                // news. Don't move it.
033                oi = new OI();
034                // Show what command your subsystem is running on the SmartDashboard
035                //SmartDashboard.putData(exampleSubsystem);
036                SmartDashboard.putData(cam);
037                SmartDashboard.putData(camServos);
038                SmartDashboard.putData(pwmDriveBase);
039        }
040
041        public CommandBase(String name) {
042                super(name);
043        }
044
045        public CommandBase() {
046                super();
047        }
048}

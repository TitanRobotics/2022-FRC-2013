005package org.usfirst.frc2022.commands;
006import org.usfirst.frc2022.Joysticks.Attack3;
007import org.usfirst.frc2022.OI;
008import org.usfirst.frc2022.subsystems.Pickup;
009import org.usfirst.frc2022.subsystems.Shooter;
010        
011
012
013/**This code controls the commands for subsystem Pickup.java and 
014 * runs the relays when button is held down
015 *
016 * @author Malachi Loviska
017 */
018public class PickupDumbCommand extends CommandBase {
019    
020    Attack3 gandalfTheAttack3;
021   
022    
023    public PickupDumbCommand() {
024
025        requires(pickup);
026       
027    }
028       /**
029         * Runs the function when the command first starts and assigns the attack 3 form OI
030         */
031
032    protected void initialize() {
033        gandalfTheAttack3 = oi.getAttack3();
034        
035    }
036
037    /**
038     * Runs the relays the command is running and 
039     * detects if the Button is being held down 
040     */
041    protected void execute() {
042        if (gandalfTheAttack3.getRawButton(6)) {
043              pickup.roll();
044        } 
045        else if (gandalfTheAttack3.getRawButton(7)){
046            pickup.drop();
047        }
048        else {pickup.stop();}
049
050    }
051    
052    protected boolean isFinished() {
053        return false;
054    }
055
056   /**
057     * This function is called when the command ends. It stops the robot.
058     *
059     * @param
060     * @return
061     */
062    protected void end() {
063        pickup.stop();
064    }
065
066    /**
067     * This function is called when the command is interrupted. It stops the
068     * robot.
069     *
070     * @param
071     * @return
072     */
073    protected void interrupted() {
074        pickup.stop();
075    }
076}

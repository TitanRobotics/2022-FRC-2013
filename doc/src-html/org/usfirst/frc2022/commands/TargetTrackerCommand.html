001package org.usfirst.frc2022.commands;
002
003import org.usfirst.frc2022.Utils;
004
005import edu.wpi.first.wpilibj.command.Command;
006import edu.wpi.first.wpilibj.image.ParticleAnalysisReport;
007import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
008
009public class TargetTrackerCommand extends CommandBase {
010
011    /*
012     * The portion of the image that the box can be in
013     */
014    public final double middle = .01;
015
016    public TargetTrackerCommand() {
017        requires(cam);
018        requires(camServos);
019    }
020
021    protected void initialize() {
022    }
023
024    protected void execute() {
025        ParticleAnalysisReport[] report = cam.analyze();
026        if (report.length < 1) {
027            SmartDashboard.putString("DEBUG","None found");
028            return;
029        }
030        ParticleAnalysisReport particle = report[0];
031        double xpos = particle.center_mass_x_normalized;
032        double ypos = particle.center_mass_y_normalized;
033
034        double rotate = Utils.sign(xpos) * 1;
035        double pitch = Utils.sign(ypos) * 1;
036
037        SmartDashboard.putString("DEBUG", "X = " + xpos + ", Y= " + ypos + "; r=" + rotate + ",p=" + pitch);
038        camServos.setRotateAngle(camServos.getRotateAngle() + rotate);
039        camServos.setPitchAngle(camServos.getRotateAngle() + pitch);
040        camServos.updateSD();
041    }
042
043    protected boolean isFinished() {
044        return false;
045    }
046
047    protected void end() {
048    }
049
050    protected void interrupted() {
051    }
052}

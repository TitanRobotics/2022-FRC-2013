001package org.usfirst.frc2022.commands;
002
003import org.usfirst.frc2022.Joysticks.Xbox360;
004
005public class Servo45 extends CommandBase {
006
007    public Servo45() {
008        requires(camServos);
009    }
010
011    protected void initialize() {
012    }
013
014    protected void execute() {
015        camServos.updateSD();
016        Xbox360 xbox = oi.getXbawks();
017        camServos.setPitchAngle(90*(xbox.GetRightY()/2+.5));
018        camServos.setRotateAngle(90*(xbox.GetRightY()/2+.5));
019    }
020
021    protected boolean isFinished() {
022        return false;
023    }
024
025    protected void end() {
026    }
027
028    protected void interrupted() {
029    }
030}

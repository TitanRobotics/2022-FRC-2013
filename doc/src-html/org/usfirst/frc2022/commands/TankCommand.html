005package org.usfirst.frc2022.commands;
006import org.usfirst.frc2022.Joysticks.Xbox360;
007/**
008 *
009 * @author Student
010 */
011public class TankCommand extends CommandBase{
012    Xbox360 xbox;
013    public TankCommand(){
014        //initializes command system, system has no dependencies
015        requires(pwmDriveBase);
016    }
017    
018    protected void initialize(){
019        pwmDriveBase.stop(); //stops the motors when the robot starts up
020    }
021    
022    protected void execute(){
023        xbox=oi.getXbawks();
024        double leftSpeed, rightSpeed;
025        leftSpeed = xbox.GetLeftY();
026        rightSpeed = xbox.GetRightY();
027        if(Math.abs(leftSpeed)<0.2) {
028            leftSpeed = 0;
029        }
030        if(Math.abs(rightSpeed)<0.2) {
031            rightSpeed = 0;
032        }
033        pwmDriveBase.drive(leftSpeed, rightSpeed); //the robot will move!
034        if (xbox.GetAValue()){
035        pwmDriveBase.flipJags(); //flips if the flip button is pushed
036        }
037    }
038
039    protected boolean isFinished() {
040        return(false); //It's not over until it's over!
041    }
042
043    protected void end() {
044        pwmDriveBase.stop(); //stop once the game ends
045    }
046
047    protected void interrupted() {
048        pwmDriveBase.stop(); //stop if everything goes crazy
049    }
050}

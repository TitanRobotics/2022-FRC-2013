015public class MecanumCommand extends CommandBase{
016
017    Xbox360 controller;
018    double speedLeftFront;
019    double speedRightFront;
020    double speedLeftBack;
021    double speedRightBack;
022    
023    public MecanumCommand(){
024        requires(pwmGeneric);
025        
026    }
027    
028    /**
029     * This function is called when the command starts.
030     * It c
031     * 
032     * @author Titan Robotics (2022)
033     * @param
034     * @return 
035     */
036    protected void initialize() {
037        controller = oi.getXbawks();
038        speedLeftFront = 0;
039        speedRightFront = 0;
040        speedLeftBack = 0;
041        speedRightBack = 0;
042    }
043
044    /**
045     * The primary loop for the mecanum drive. Gets controller input
046     * and does sends commands to PWM_Generic.java to move the bot.
047     * It also contains the mecanum algorithms.
048     * 
049     * @author Titan Robotics (2022)
050     * @param
051     * @return 
052     */
053    protected void execute() {
054        
055        double direction = (controller.GetLeftAngle(false) * 180) / Math.PI;
056        double magnitude = (controller.GetLeftMagnitude() / 2);
057        
058        double rotation = controller.GetRightX() / 3; //silverStick->GetTwist();
059        
060        //Set Deadband on Translation and Rotation
061        if(magnitude < 0.2) {
062            magnitude = 0;
063        }
064        if((Math.abs(rotation) * 3) < 0.2) {
065            rotation = 0;
066        }
067        
068        if(controller.GetTriggers() > .5) {
069            magnitude /= 2;
070        }
071        if(controller.GetTriggers() < -.5) {
072            magnitude *= 2;
073        }
074        
075        double cosD = Math.cos((direction + 45.0) * Math.PI / 180.0);
076        double sinD = Math.cos((direction - 45.0) * Math.PI / 180.0);
077
078        speedLeftFront = Utils.clamp((sinD * magnitude + rotation) * 2,1,-1);
079        speedLeftBack = Utils.clamp((cosD * magnitude + rotation) * 2,1,-1);
080        speedRightFront = Utils.clamp((cosD * magnitude - rotation) * 2,1,-1);
081        speedRightBack = Utils.clamp((sinD * magnitude - rotation) * 2,1,-1);
082                        
083        //Update Mecanum Subsystem
084        pwmGeneric.driveMecanum(-speedLeftFront, -speedRightFront, speedLeftBack, speedRightBack);
085        
086    }
087
088    protected boolean isFinished() {
089        return false;
090    }
091    
092    /**
093     * This function is called when the command ends.

097     * @return 
098     */

100        pwmGeneric.stop();
101    }

105     *

107     * @param
108     * @return 
109     */
110    protected void interrupted() {
111        pwmGeneric.stop();
112    }

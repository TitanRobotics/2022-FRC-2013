005package org.usfirst.frc2022.commands;

007import org.usfirst.frc2022.Joysticks.Xbox360;
008import org.usfirst.frc2022.Utils;

010
011/**
012 * This is the command for use with a mecanum drivebase.
013 * It contains all controls and functions needed to 
014 * run a mecanum robot.
015 * 
016 * @author Titan Robotics (2022)
017 * @author Michael Hrcek
018 */
019public class MecanumCommand extends CommandBase{
020
021    //declare variables
022    Xbox360 controller;
023    double speedLeftFront;
024    double speedRightFront;
025    double speedLeftBack;
026    double speedRightBack;
027    
028    /**
029     * The constructor. It requires pwmDriveBase from CommandBase.java
030     * 
031     * @param
032     * @return
033     */
034    public MecanumCommand(){
035        requires(pwmDriveBase);
036        
037    }
038    
039    /**
040     * This function is called when the command starts.
041     * It registers the controller and sets the speeds
042     * to zero.
043     * 
044     * @param
045     * @return 
046     */
047    protected void initialize() {
048        controller = oi.getXbawks();
049        speedLeftFront = 0;
050        speedRightFront = 0;
051        speedLeftBack = 0;
052        speedRightBack = 0;
053    }
054
055    /**
056     * The primary loop for the mecanum command. Gets controller input
057     * and commands the generic PWM to move the bot.
058     * 
059     * @param
060     * @return 
061     */
062    protected void execute() {
063        
064        //The direction and magnitude and rotation
065        double direction = (controller.GetLeftAngle(false) * 180) / Math.PI;
066        double magnitude = (controller.GetLeftMagnitude() / 2);
067        double rotation = controller.GetRightX() / 3;
068        
069        //Set Deadband on Translation and Rotation
070        if(magnitude < 0.2) {
071            magnitude = 0;
072        }
073        if((Math.abs(rotation) * 3) < 0.2) {
074            rotation = 0;
075        }
076        
077        if(controller.GetTriggers() > .5) {
078            magnitude /= 2;
079        }
080        if(controller.GetTriggers() < -.5) {
081            magnitude *= 2;
082        }
083        
084        //define custom cosine and sine functions
085        double cosD = Math.cos((direction + 45.0) * Math.PI / 180.0);
086        double sinD = Math.cos((direction - 45.0) * Math.PI / 180.0);
087
088        //set speeds and clamp at 1 and -1
089        speedLeftFront = Utils.clamp((sinD * magnitude - rotation) * 2,1,-1);
090        speedLeftBack = Utils.clamp((cosD * magnitude + rotation) * 2,1,-1);
091        speedRightFront = Utils.clamp((cosD * magnitude - rotation) * 2,1,-1);
092        speedRightBack = Utils.clamp((sinD * magnitude + rotation) * 2,1,-1);
093                        
094        //Update Mecanum Subsystem
095        if(controller.GetAValue()){pwmDriveBase.driveMecanum(speedRightBack, speedLeftBack, -speedRightFront, -speedLeftFront);} 
096        else {pwmDriveBase.driveMecanum(-speedLeftFront, -speedRightFront, speedLeftBack, speedRightBack);}
097        
098        
099    }
100
101    protected boolean isFinished() {
102        return false;
103    }
104    
105    /**
106     * This function is called when the command ends.
107     * It stops the robot.
108     * 
109     * @param
110     * @return 
111     */
112    protected void end() {
113        pwmDriveBase.stop();
114    }
115
116    /**
117     * This function is called when the command is interrupted.
118     * It stops the robot.
119     * 
120     * @param
121     * @return 
122     */
123    protected void interrupted() {
124        pwmDriveBase.stop();
125    }

011/**
012 *
013 * @author Michael
014 */
015public class MecanumCommand extends CommandBase{
016
017    //declare variables
018    Xbox360 controller;
019    double speedLeftFront;
020    double speedRightFront;
021    double speedLeftBack;
022    double speedRightBack;
023    
024    /**
025     * The constructor. It requires pwmGeneric from CommandBase.java
026     * 
027     * @author Titan Robotics (2022)
028     * @param
029     * @return
030     */
031    public MecanumCommand(){
032        requires(pwmGeneric);
033        
034    }
035    
036    /**
037     * This function is called when the command starts.
038     * It registers the controller and sets the speeds.
039     * 
040     * @author Titan Robotics (2022)
041     * @param
042     * @return 
043     */
044    protected void initialize() {
045        controller = oi.getXbawks();
046        speedLeftFront = 0;
047        speedRightFront = 0;
048        speedLeftBack = 0;
049        speedRightBack = 0;
050    }
051
052    /**
053     * The primary loop for the mecanum drive. Gets controller input
054     * and does sends commands to PWM_Generic.java to move the bot.
055     * It also contains the mecanum algorithms.
056     * 
057     * @author Titan Robotics (2022)
058     * @param
059     * @return 
060     */
061    protected void execute() {
062        
063        //The direction and magnitude and rotation
064        double direction = (controller.GetLeftAngle(false) * 180) / Math.PI;
065        double magnitude = (controller.GetLeftMagnitude() / 2);
066        double rotation = controller.GetRightX() / 3;
067        
068        //Set Deadband on Translation and Rotation
069        if(magnitude < 0.2) {
070            magnitude = 0;
071        }
072        if((Math.abs(rotation) * 3) < 0.2) {
073            rotation = 0;
074        }
075        
076        if(controller.GetTriggers() > .5) {
077            magnitude /= 2;
078        }
079        if(controller.GetTriggers() < -.5) {
080            magnitude *= 2;
081        }
082        
083        //define custom cosine and sine functions
084        double cosD = Math.cos((direction + 45.0) * Math.PI / 180.0);
085        double sinD = Math.cos((direction - 45.0) * Math.PI / 180.0);
086
087        //set speeds and clamp at 1 and -1
088        speedLeftFront = Utils.clamp((sinD * magnitude + rotation) * 2,1,-1);
089        speedLeftBack = Utils.clamp((cosD * magnitude + rotation) * 2,1,-1);
090        speedRightFront = Utils.clamp((cosD * magnitude - rotation) * 2,1,-1);
091        speedRightBack = Utils.clamp((sinD * magnitude - rotation) * 2,1,-1);
092                        
093        //Update Mecanum Subsystem
094        pwmGeneric.driveMecanum(-speedLeftFront, -speedRightFront, speedLeftBack, speedRightBack);
095        
096    }
097
098    protected boolean isFinished() {
099        return false;
100    }
101    
102    /**
103     * This function is called when the command ends.
104     * 
105     * @author Titan Robotics (2022)
106     * @param
107     * @return 
108     */
109    protected void end() {
110        pwmGeneric.stop();
111    }
112
113    /**
114     * This function is called when the command is interrupted.
115     * 
116     * @author Titan Robotics (2022)
117     * @param
118     * @return 
119     */
120    protected void interrupted() {
121        pwmGeneric.stop();
122    }

005package org.usfirst.frc2022.commands;
006
007import org.usfirst.frc2022.Joysticks.Attack3;
008import org.usfirst.frc2022.Utils;
009
010/**
011 *
012 * @author Emma Sloan
013 */
014public class RotationPitchCommand extends CommandBase {
015    Attack3 attack3;
016    
017    public RotationPitchCommand() {
018        // Use requires() here to declare subsystem dependencies
019        // eg. requires(chassis);
020        requires(shooterPitch);
021        requires(shooterRotation);
022        attack3 = oi.getAttack3();
023    }
024
025    // Called just before this Command runs the first time
026    protected void initialize() {
027        shooterPitch.disable(); //Disables the PID loop for the pitch
028        shooterRotation.disable(); //Disables the PID loop for the rotation
029        shooterPitch.setPitch(0);
030        shooterRotation.setRotation(0);
031    }
032
033    // Called repeatedly when this Command is scheduled to run
034    protected void execute() {
035        //manual command to set the pitch to the attack 3 y axis value        
036        shooterPitch.setPitch(Utils.clamp(attack3.GetY(), 0.75, -0.75));
037        //manual command to set the rotation to the attack 3 x axis value
038        shooterRotation.setRotation(Utils.clamp(attack3.GetX(), 0.75, -0.75));
039        
040    }
041
042    // Make this return true when this Command no longer needs to run execute()
043    protected boolean isFinished() {
044        return false;
045    }
046
047    // Called once after isFinished returns true
048    protected void end() {
049    }
050
051    // Called when another command which requires one or more of the same
052    // subsystems is scheduled to run
053    protected void interrupted() {
054        shooterPitch.setPitch(0);
055        shooterRotation.setRotation(0);
056    }
057}

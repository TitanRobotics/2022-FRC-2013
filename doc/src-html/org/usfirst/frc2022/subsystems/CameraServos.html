001package org.usfirst.frc2022.subsystems;
002
003import edu.wpi.first.wpilibj.Servo;
004import edu.wpi.first.wpilibj.command.Subsystem;
005import edu.wpi.first.wpilibj.livewindow.LiveWindow;
006import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
007import org.usfirst.frc2022.RobotMap;
008import org.usfirst.frc2022.commands.Servo45;
009
010/**
011 * This subsystem is for use with the camera on Clifford. It uses two servos,
012 * one for changing the camera's pitch and another for the camera's rotation. *
013 */
014public class CameraServos extends Subsystem {
015
016    private Servo pitchServo;
017    private Servo rotateServo;
018
019    /**
020     * Creates a new CameraServos subsystem that assumes the default digital
021     * servos module using channels in the RobotMap
022     *
023     */
024    public CameraServos() {
025        pitchServo = new Servo(RobotMap.pitchServo);
026        rotateServo = new Servo(RobotMap.rotateServo);
027    }
028
029    /**
030     * Creates a new CameraServos subsystem that assumes the default digital
031     * servos module.
032     *
033     * @param pitchChannel - channel for the pitch servo
034     * @param rotateChannel - channel for the rotate servo
035     *
036     */
037    public CameraServos(int pitchChannel, int rotateChannel) {
038        pitchServo = new Servo(pitchChannel);
039        rotateServo = new Servo(rotateChannel);
040    }
041
042    /**
043     * Creates a new CameraServos subsystem that specifies the digital servo
044     * module.
045     *
046     * @param pitchSlot - slot for the pitch servo
047     * @param pitchChannel - channel for the pitch servo
048     * @param rotateSlot - slot for the rotate servo
049     * @param rotateChannel - channel for the rotate servo
050     *
051     */
052    public CameraServos(int pitchSlot, int pitchChannel, int rotateSlot, int rotateChannel) {
053        pitchServo = new Servo(pitchSlot, pitchChannel);
054        rotateServo = new Servo(rotateSlot, rotateChannel);
055        /*setupLiveWindow();
056         LiveWindow.setEnabled(true);
057         LiveWindow.run();*/
058
059    }
060
061    /**
062     * Sets the angle for the pitch servo
063     *
064     * @param angle - angle to pitch
065     *
066     */
067    public void setPitchAngle(double angle) {
068        angle = angle / 90.0;
069        if (angle >= .95) {
070            angle = .95;
071        } else if (angle <= 0.05) {
072            angle = 0.05;
073        }
074        pitchServo.set(angle);
075        //System.out.println(angle);
076    }
077
078    /**
079     * Sets the angle for the rotate servo
080     *
081     * @param angle - angle to rotate
082     *
083     */
084    public void setRotateAngle(double angle) {
085
086        angle = angle / 90.0;
087        if (angle >= .95) {
088            angle = .95;
089        } else if (angle <= 0.05) {
090            angle = 0.05;
091        }
092        rotateServo.set(angle);
093        //System.out.println(angle);
094    }
095
096    /**
097     * Gives the angle for the pitch servo
098     *
099     * @return getAngle() - angle of the pitch servo
100     *
101     */
102    public double getPitchAngle() {
103        return pitchServo.get() * 90;
104    }
105
106    /**
107     * Gives the angle for the rotate servo
108     *
109     * @return getAngle() - angle of the rotate servo
110     *
111     */
112    public double getRotateAngle() {
113        return rotateServo.get() * 90;
114    }
115
116    public void updateSD() {
117        SmartDashboard.putNumber("Camera rotation", this.getRotateAngle());
118        SmartDashboard.putNumber("Camera pitch", this.getPitchAngle());
119    }
120
121    public void setupLiveWindow() {
122        LiveWindow.addActuator("CameraServos", "Pitch Servo", pitchServo);
123        LiveWindow.addActuator("CameraServos", "Rotate Servo", rotateServo);
124    }
125
126    /**
127     * Default command for the CameraServos subsystem
128     */
129    protected void initDefaultCommand() {
130        //setDefaultCommand(new Servo45());
131        setDefaultCommand(new Servo45());
132    }
133}

001package org.usfirst.frc2022.subsystems;
002
003import edu.wpi.first.wpilibj.command.Subsystem;
004import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
005import edu.wpi.first.wpilibj.Servo;
006import org.usfirst.frc2022.RobotMap;
007import org.usfirst.frc2022.commands.Servo45;
008import org.usfirst.frc2022.commands.TargetTrackerCommand;
009
010/**
011 * This subsystem is for use with the camera on Clifford. It uses two
012 * servos, one for changing the camera's pitch and another for the 
013 * camera's rotation. *
014 */
015
016public class CameraServos extends Subsystem {
017        
018        private Servo pitchServo;
019        private Servo rotateServo;
020        
021        /**
022         * Creates a new CameraServos subsystem that assumes the default
023         * digital servos module using channels in the RobotMap
024         * 
025         */
026        public CameraServos(){
027                pitchServo = new Servo(RobotMap.pitchServo);
028                rotateServo = new Servo(RobotMap.rotateServo);          
029        }
030        
031        /**
032         * Creates a new CameraServos subsystem that assumes the default
033         * digital servos module.
034         * 
035         * @param pitchChannel - channel for the pitch servo
036         * @param rotateChannel - channel for the rotate servo
037         * 
038         */
039        public CameraServos(int pitchChannel, int rotateChannel){
040                pitchServo = new Servo(pitchChannel);
041                rotateServo = new Servo(rotateChannel);         
042        }
043        
044        /**
045         * Creates a new CameraServos subsystem that specifies the 
046         * digital servo module.
047         * 
048         * @param pitchSlot - slot for the pitch servo
049         * @param pitchChannel - channel for the pitch servo
050         * @param rotateSlot - slot for the rotate servo
051         * @param rotateChannel - channel for the rotate servo
052         * 
053         */
054        public CameraServos(int pitchSlot, int pitchChannel, int rotateSlot, int rotateChannel){
055                pitchServo = new Servo(pitchSlot, pitchChannel);
056                rotateServo = new Servo(rotateSlot, rotateChannel);
057                
058        }
059        
060        /**
061         * Sets the angle for the pitch servo
062         * 
063         * @param angle - angle to pitch
064         * 
065         */
066        public void setPitchAngle(double angle){
067                if(angle > 90.0){
068                        angle = 90.0;
069                }
070                else if(angle < 0.0){
071                        angle = 0.0;
072                }
073                pitchServo.setAngle(angle);
074                System.out.print(angle);
075        }
076        
077        /**
078         * Sets the angle for the rotate servo
079         * 
080         * @param angle - angle to rotate
081         * 
082         */
083        public void setRotateAngle(double angle){
084                if(angle > 90.0){
085                        angle = 90.0;
086                }
087                else if(angle < 0.0){
088                        angle = 0.0;
089                }
090                rotateServo.setAngle(angle);
091                System.out.print(angle);
092        }
093        
094        /**
095         * Gives the angle for the pitch servo
096         * 
097         * @return getAngle() - angle of the pitch servo
098         * 
099         */
100        public double getPitchAngle(){
101                return pitchServo.getAngle();
102        }
103        
104        /**
105         * Gives the angle for the rotate servo
106         * 
107         * @return getAngle() - angle of the rotate servo
108         * 
109         */
110        public double getRotateAngle(){
111                return rotateServo.getAngle();
112        }
113        
114        public void updateSD(){
115                SmartDashboard.putDouble("Camera rotation", this.getRotateAngle());
116                SmartDashboard.putDouble("Camera pitch", this.getPitchAngle());
117        }
118        
119        /**
120         * Default command for the CameraServos subsystem
121         */
122        protected void initDefaultCommand() {
123                setDefaultCommand(new Servo45());
124        }
125
126}

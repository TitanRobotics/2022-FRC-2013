001package org.usfirst.frc2022.subsystems;
002
003import edu.wpi.first.wpilibj.Jaguar;
004
005public interface Drive_Generic {
006
007    /**
008     * Set the speed of the motors on the left side of the robot.
009     *
010     * @param speed The speed at which the motors move
011     * @returns
012     */
013    void setLeft(double speed);
014
015    /**
016     * Set the speed of the motors on the right side of the robot.
017     *
018     * @param speed The speed at which the motors move
019     * @returns
020     */
021    void setRight(double speed);
022
023    /**
024     * Move the robot. Only go forward and back.
025     *
026     * @param speed The speed of the motors
027     * @returns
028     */
029    void drive(double speed);
030
031    /**
032     * Move the robot. Each side can have a different speed to allow for
033     * turning.
034     *
035     * @param speedLeft The speed of the left motors
036     * @param speedRight The speed of the right motors
037     * @returns
038     */
039    void drive(double speedLeft, double speedRight);
040
041    /**
042     * Halt the motors
043     *
044     * @param
045     * @return
046     */
047    void stop();
048
049    /**
050     * Function that checks if a number is even. Specifically used for checking
051     * the size of arrays.
052     *
053     * @param size
054     * @return whether or not the number passed through is even
055     */
056    boolean checkEven(int size);
057
058    /**
059     * Separate the jaguars to left and right sides (odds left, evens right)
060     *
061     * @param jags
062     * @return
063     */
064    void seperateJags(Jaguar[] jags);
065
066    /**
067     * Flips the jaguars on each side; then switches the sides
068     *
069     * @param
070     * @return
071     */
072    void flipJags();
073}

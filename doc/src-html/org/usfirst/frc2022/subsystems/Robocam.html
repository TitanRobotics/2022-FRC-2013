016/**
017 * Image processor using NIVision to find "shining" rectangles once an image is
018 * filtered with low light. The CriteriaCollection contains the "rules" used to
019 * analyze the maximum width of the received rectangles.
020 *
021 * Modified from original post on
022 * http://www.chiefdelphi.com/forums/showthread.php?t=109657
023 *
024 * @author FRC Team #3381 (original), FRC Team #2022 (modified)
025 * @param ip - String of the camera's IP address
026 *
027 */
028public class Robocam extends Subsystem {
029
030    private AxisCamera camera;                          //camera instance
031    private CriteriaCollection collection;      //criteria for analyzing image
032
033    /**
034     * Creates a new Robocam instance with a minimum width of 30 px and a
035     * maximum width of 400 px as well as a minimum height of 40 px and maximum
036     * height of 400 px.
037     *
038     * @author FRC Team #3381 (original), FRC Team #2022 (modified)
039     * @param ip - String of the camera's IP address
040     * @return
041     */
042    public Robocam(String ip) {
043        camera = AxisCamera.getInstance(ip);
044        collection = new CriteriaCollection();
045        collection.addCriteria(MeasurementType.IMAQ_MT_BOUNDING_RECT_WIDTH, 30, 400, false);
046        collection.addCriteria(MeasurementType.IMAQ_MT_BOUNDING_RECT_HEIGHT, 40, 400, false);
047    }
048
049    /**
050     * Analyzes an image taken by a robot's camera, using BinaryImage objects to
051     * refine the image, resulting in the center of each rectangle found, if any
052     *
053     * @author FRC Team #3381 (original), FRC Team #2022 (modified)
054     * @param
055     * @return
056     *
057     */
058    public ParticleAnalysisReport[] analyze() {
059        ParticleAnalysisReport[] finalReport;
060
061        try {
062            ColorImage image = camera.getImage();
063            //newRGBImage(string filename) for loading an image
064            //image.thresholdHSL(100, 120, saturationLow, saturationHigh, luminenceLow, luminenceHigh)
065            //BinaryImage thresholdImage = image.thresholdRGB(0, 88, 25, 255, 0, 47);           // keep green reflection
066            BinaryImage thresholdImage = image.thresholdHSL(43, 66, 100, 256, 0, 250);
067            BinaryImage bigObjectsImage = thresholdImage.removeSmallObjects(false, 2);  // remove small artifacts
068            BinaryImage convexHullImage = bigObjectsImage.convexHull(false);            // fill in occluded rectangles
069            BinaryImage filteredImage = convexHullImage.particleFilter(collection);     // find filled in rectangles
070
071            ParticleAnalysisReport[] reports = filteredImage.getOrderedParticleAnalysisReports();  // get list of results
072            finalReport = reports;
073            SmartDashboard.putNumber("Number of artifacts found: ", reports.length);            //Tell Drivers Station artifacts found
074            for (int i = 0; i < reports.length; i++) {                                       // print results
075                ParticleAnalysisReport r = reports[i];
076
077                SmartDashboard.putNumber("Artifact :             ", i);
078                SmartDashboard.putNumber("Center (x) normalized: ", r.center_mass_x_normalized);
079                SmartDashboard.putNumber("Center (y) normalized: ", r.center_mass_y_normalized);
080                SmartDashboard.putNumber("Height:                ", r.imageHeight);
081                SmartDashboard.putNumber("Width:                 ", r.imageWidth);
082            }
083
084            /* MUST USE FREE FUNCTION: all images are currently allocated in C structures */
085            filteredImage.free();
086            convexHullImage.free();
087            bigObjectsImage.free();
088            thresholdImage.free();
089            image.free();
090        } //end analyze()
091        catch (AxisCameraException e) {
092            finalReport = null;
093            e.printStackTrace();
094        } catch (NIVisionException ex) {
095            finalReport = null;
096            ex.printStackTrace();
097        }
098
099        return finalReport;
100    }
101
102    /**
103     * Default command for the Robocam subsystem
104     */
105    protected void initDefaultCommand() {
106    }
107}

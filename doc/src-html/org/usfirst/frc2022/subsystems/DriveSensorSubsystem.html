001package org.usfirst.frc2022.subsystems;
002
003import edu.wpi.first.wpilibj.*;
004import edu.wpi.first.wpilibj.command.Subsystem;
005import org.usfirst.frc2022.RobotMap;
006
007
008/**
009 *Subsystem created to get the distance and speed the wheels are moving in Autonomous
010 * @author Malachi Loviska
011 */
012public class DriveSensorSubsystem extends Subsystem {
013
014    Encoder Drive_Encoder1;
015    Encoder Drive_Encoder2;
016    Gyro Gyro_Sensor1;
017
018    /*
019     * Initialize  Encoders 
020     */
021    /**
022     * Constructs the Jaguar, Encoder, and Solenoid and assigns ports from robot
023     * map. It also starts the Encoder
024     */
025    public DriveSensorSubsystem() {
026
027
028
029        Drive_Encoder1 = new Encoder(RobotMap.DriveEncoder1[5],
030                RobotMap.DriveEncoder1[6]);
031        Drive_Encoder2 = new Encoder(RobotMap.DriveEncoder2[7],
032                RobotMap.DriveEncoder1[8]);
033        Gyro_Sensor1.reset();
034
035
036
037
038        this.init();
039
040    }
041    
042    public void ResetGyro()
043    {
044        Gyro_Sensor1.reset();
045    }
046    
047    public double GetGyroAngle()
048    {
049        return Gyro_Sensor1.getAngle();
050    }
051    
052    public void initDefaultCommand(){
053        
054    }
055
056    public void init() {
057
058        Drive_Encoder1.setDistancePerPulse(1.0472 / 360);
059        Drive_Encoder2.setDistancePerPulse(1.0472 / 360);
060        Drive_Encoder1.start();
061        Drive_Encoder2.start();
062    }
063
064    public void reset() {
065
066        Drive_Encoder1.reset();
067        Drive_Encoder2.reset();
068    }
069
070    public double getRightSpeed() {
071        return Drive_Encoder1.getRate();
072
073    }
074
075    public double getLeftSpeed() {
076        return Drive_Encoder2.getRate();
077    }
078
079    public double GetRightDistance() {
080        return Drive_Encoder1.getDistance();
081    }
082
083    public double GetLeftDistance() {
084        return Drive_Encoder2.getDistance();
085    }
086}

004import edu.wpi.first.wpilibj.Jaguar;
005import edu.wpi.first.wpilibj.command.Subsystem;

009/** 
010 * 
011 * Generic PWM subsystem used for the drivebase.
012 * 
013 * @author Titan Robotics (2022)
014 * @author Michael Hrcek
015 * 
016 */
017public class PWM_Generic extends Subsystem implements Drive_Generic{
018        
019        Jaguar[] jags;          //Array of a variable size that holds the jaguars
020        Jaguar[] jagsLeft;      //jaguars on the left side of the robot
021        Jaguar[] jagsRight;     //jaguars on the right side of the robot
022        
023        /**
024         * Constructor for the subsystem. Check for an even number of jaguars.
025         * If there are an even number of jaguars, use them. Otherwise, tell
026         * the driver to add more jaguars.
027         * 
028         * @param jags Array of jaguars used for controlling the drivebase
029         * @return
030         */
031        public PWM_Generic(Jaguar[] newJags){ //Check if there are an even number of jaguars
032        
033            if(checkEven(newJags.length)){
034                this.jags = newJags;
035                seperateJags(this.jags);
036                //assignLiveWindow();
037            } else {System.out.println("Odd number of Jaguars");}       
038        }
039        
040        /**
041         * Constructor for the subsystem. Check for an even number of ports.
042         * If there are an even number of ports, create new jaguars at those ports.
043         * Otherwise, tell the driver to assign an even number of ports to the robot.
044         * 
045         * @param ports Array of ints storing the ports used by the Jaguars
046         * @return
047         */
048        public PWM_Generic(int[] ports){ //Check if there are an even number of jaguars
049            Jaguar[] newJags = new Jaguar[ports.length];
050            if(checkEven(ports.length)){
051                
052                for(int i=0; i<ports.length;i++){
053                    newJags[i] = new Jaguar(ports[i]);
054                }
055                
056                this.jags = newJags;
057                seperateJags(this.jags);
058                //assignLiveWindow();
059            } else {System.out.println("Odd number of Jaguars");}       
060        }
061        
062        /**
063         * This function initiates the default command.
064         * 
065         * @param
066         * @returns
067         */
068        protected void initDefaultCommand() {
069                setDefaultCommand(new MecanumCommand());
070        }
071        
072        /**
073         * Set the speed of the motors on the left side of
074         * the robot.
075         * 
076         * @param speed The speed at which the motors move
077         * @returns
078         */
079        public void setLeft(double speed){
080                for(int i=0; jagsLeft.length > i; i++){
081                        jagsLeft[i].set(Utils.clamp(speed,1,-1));
082                }
083        }
084        
085        /**
086         * Set the speed of the motors on the right side of
087         * the robot.
088         * 
089         * @param speed The speed at which the motors move
090         * @returns
091         */
092        public void setRight(double speed){
093                for(int i=0; jagsRight.length > i; i++){
094                        jagsRight[i].set(Utils.clamp(-speed,1,-1));
095                }
096        }
097        
098        /**
099         * Set the speed of all the motors. Only allows for
100         * forward and backward movement.
101         * 
102         * @param speed The speed of the motors
103         * @returns
104         */
105        public void drive(double speed){
106                setRight(speed);
107                setLeft(-speed);
108        }
109        
110        /**
111         * Set the speed of the motors. Each side can be
112         * set to a different speed. Useful for a tank drive.
113         * 
114         * @param speedLeft The speed of the left motors
115         * @param speedRight The speed of the right motors
116         * @returns
117         */
118        public void drive(double speedLeft, double speedRight){
119                setRight(speedRight);
120                setLeft(speedLeft);
121        }
122        
123        /**
124         * Designed for use in mecanum drives. Set the 
125         * speed for each motor. Must have exactly 4 motors.
126         * 
127         * @param speedLeftFront Speed of the front left wheel
128         * @param speedRightFront Speed of the front right wheel
129         * @param speedLeftBack Speed of the back left wheel
130         * @param speedRightBack Speed of the back right wheel
131         * @return 
132         */
133        public void driveMecanum(double speedLeftFront, double speedRightFront,double speedLeftBack, double speedRightBack){
134                jagsLeft[0].set(speedLeftFront);
135                jagsLeft[1].set(speedLeftBack);
136                jagsRight[0].set(speedRightFront);
137                jagsRight[1].set(speedRightBack);
138        }
139        
140        /**
141         * Force the motors to stop.
142         * 
143         * @param
144         * @return
145         */
146        public void stop(){
147                setRight(0);
148                setLeft(0);
149        }
150        
151        /**
152         * Function that checks if a number is even. Specifically
153         * used for checking the size of arrays.
154         * 
155         * @param size
156         * @return whether or not the number passed through is even
157         */
158        public boolean checkEven(int size){
159                if((size % 2)==1){return false;} else {return true;}
160        } //end private boolean checkEven(int size)
161        
162        
163        
164        /**
165         * Separate the jaguars to left and right sides with odd numbered
166         * jaguars being assigned to the left and even jaguars assigned on
167         * the right.
168         * 
169         * @param jags
170         * @return
171         */
172        public void seperateJags(Jaguar[] jags){
173                jagsLeft = new Jaguar[jags.length/2];
174                jagsRight = new Jaguar[jags.length/2];
175                for(int i=0,j=0,k=0; i< jags.length; i++){
176                        if(checkEven(jags[i].getChannel())){
177                                jagsRight[j]=jags[i];
178                                j++;
179                        } else if (!checkEven(jags[i].getChannel())){
180                                jagsLeft[k]=jags[i];
181                                k++;
182                        }
183                }
184        }
185        
186        /**
187         * Allows the front of the robot to become the back
188         * by switching the jaguars. Useful for tank drives.
189         * 
190         * @param
191         * @return
192         */
193        public void flipJags(){
194            
195                int length = jagsLeft.length/2;
196                Jaguar[] tempJags;
197                
198                for(int i=0; i < length; i++){ 
199                        Jaguar temp = jagsLeft[i];
200                        jagsLeft[i] = jagsLeft[length-1-i];
201                        jagsLeft[length-1-i] = temp;
202                }
203                
204                for(int i=0; i < length; i++){ 
205                        Jaguar temp = jagsRight[i];
206                        jagsRight[i] = jagsRight[length-1-i];
207                        jagsRight[length-1-i] = temp;
208                }
209                
210                tempJags=jagsRight;
211                jagsRight = jagsLeft;
212                jagsLeft = tempJags;
213                
214                //  Does not work
215                //assignLiveWindow();
216        }

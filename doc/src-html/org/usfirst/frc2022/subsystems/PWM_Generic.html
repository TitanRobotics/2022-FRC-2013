001package org.usfirst.frc2022.subsystems;
002
003import edu.wpi.first.wpilibj.command.Command;

011/** 
012 * 
013 * Generic
014 * 
015 */
016public class PWM_Generic extends Subsystem implements Drive_Generic{
017        
018        Jaguar[] jags;          //Array of a variable size that holds the jaguars
019        Jaguar[] jagsLeft;      //jaguars on the left side of the robot
020        Jaguar[] jagsRight;     //jaguars on the right side of the robot
021        
022        /**
023         * Constructor for the subsystem. Assigns jaguars if there are an
024         * even amount and there are the same number of ports.
025         * 
026         * @param jags Array of jaguars used for controlling the drivebase
027         * @return
028         */
029        public PWM_Generic(Jaguar[] newJags){ //Check if there are an even number of jaguars
030        
031            if(checkEven(newJags.length)){
032                this.jags = newJags;
033                seperateJags(this.jags);
034                //assignLiveWindow();
035            } else {System.out.println("Odd number of Jaguars");}       
036        }
037        
038        /**
039         * Constructor for the subsystem. Assigns jaguars if there are an
040         * even amount of ports.
041         * 
042         * @param ports Array of ints storing the ports used by the Jaguars
043         * @return
044         */
045        public PWM_Generic(int[] ports){ //Check if there are an even number of jaguars
046            Jaguar[] newJags = new Jaguar[ports.length];
047            if(checkEven(ports.length)){
048                
049                for(int i=0; i<ports.length;i++){
050                    newJags[i] = new Jaguar(ports[i]);
051                }
052                
053                this.jags = newJags;
054                seperateJags(this.jags);
055                //assignLiveWindow();
056            } else {System.out.println("Odd number of Jaguars");}       
057        }
058        
059        /**
060         * This function initiates the default command.
061         * 
062         * @param
063         * @returns
064         */
065        protected void initDefaultCommand() {
066                setDefaultCommand(new MecanumCommand());
067        }
068        
069        /**
070         * Set the speed of the motors on the left side of
071         * the robot.
072         * 
073         * @param speed The speed at which the motors move
074         * @returns
075         */
076        public void setLeft(double speed){
077                for(int i=0; jagsLeft.length > i; i++){
078                        jagsLeft[i].set(Utils.clamp(speed,1,-1));
079                }
080        }
081        
082        /**
083         * Set the speed of the motors on the right side of
084         * the robot.
085         * 
086         * @param speed The speed at which the motors move
087         * @returns
088         */
089        public void setRight(double speed){
090                for(int i=0; jagsRight.length > i; i++){
091                        jagsRight[i].set(Utils.clamp(speed,1,-1));
092                }
093        }
094        
095        /**
096         * Move the robot. Only go forward and back.
097         * 
098         * @param speed The speed of the motors
099         * @returns
100         */
101        public void drive(double speed){
102                setRight(speed);
103                setLeft(speed);
104        }
105        
106        /**
107         * Move the robot. Each side can have a different
108         * speed to allow for turning.
109         * 
110         * @param speedLeft The speed of the left motors
111         * @param speedRight The speed of the right motors
112         * @returns
113         */
114        public void drive(double speedLeft, double speedRight){
115                setRight(speedRight);
116                setLeft(speedLeft);
117        }
118        
119        public void driveMecanum(double speedLeftFront, double speedRightFront,double speedLeftBack, double speedRightBack){
120                jagsLeft[0].set(speedLeftFront);
121                jagsLeft[1].set(speedLeftBack);
122                jagsRight[0].set(speedRightFront);
123                jagsRight[1].set(speedRightBack);
124        }
125        
126        /**
127         * Halt the motors
128         * 
129         * @param
130         * @return
131         */
132        public void stop(){
133                setRight(0);
134                setLeft(0);
135        }
136        
137        /**
138         * Function that checks if a number is even. Specifically
139         * used for checking the size of arrays.
140         * 
141         * @param size
142         * @return whether or not the number passed through is even
143         */
144        public boolean checkEven(int size){
145                if((size % 2)==1){return false;} else {return true;}
146        } //end private boolean checkEven(int size)
147        
148        
149        
150        /**
151         * Separate the jaguars to left and right sides (odds left, evens right)
152         * 
153         * @param jags
154         * @return
155         */
156        public void seperateJags(Jaguar[] jags){
157                jagsLeft = new Jaguar[jags.length/2];
158                jagsRight = new Jaguar[jags.length/2];
159                for(int i=0,j=0,k=0; i< jags.length; i++){
160                        if(checkEven(jags[i].getChannel())){
161                                jagsRight[j]=jags[i];
162                                j++;
163                        } else if (!checkEven(jags[i].getChannel())){
164                                jagsLeft[k]=jags[i];
165                                k++;
166                        }
167                }
168        }
169        
170        /**
171         * Flips the jaguars on each side; then switches the sides
172         * 
173         * @param
174         * @return
175         */
176        public void flipJags(){
177                int length = jagsLeft.length/2;
178                Jaguar[] tempJags = new Jaguar[length];
179                
180                tempJags=jagsLeft;
181                for(int i=0; i < length; i++){ 
182                        jagsLeft[i]=tempJags[length-1-i]; //Flip jaguars on left
183                }
184                for(int i=0; i < length; i++){
185                        tempJags[i]=jagsRight[length-1-i]; //Flip jaguars on right
186                }
187                
188                jagsRight = jagsLeft;
189                jagsLeft = tempJags;
190                //assignLiveWindow();
191        }

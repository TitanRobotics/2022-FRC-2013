001package org.usfirst.frc2022;
002
003import org.usfirst.frc2022.commands.AutoAimCommand;
004
005import edu.wpi.first.wpilibj.buttons.JoystickButton;
006import org.usfirst.frc2022.Joysticks.Attack3;
007import org.usfirst.frc2022.Joysticks.Xbox360;
008import org.usfirst.frc2022.commands.HandlingCommand;
009import org.usfirst.frc2022.commands.InjectionCommand;
010import org.usfirst.frc2022.commands.SweetSpotCommand;
011
012public class OI {
013
014    // Joystick Declarations
015    // @see org.usfirst.frc2022.Joysticks
016    private Xbox360 xbawks;
017    private Attack3 attack;
018    
019    // Joystick Button Declarations
020    // Map these buttons to commands
021    private JoystickButton targetTrackerButton;
022    private JoystickButton injectorButton;
023    private JoystickButton specialSweetSpotButton;
024    private JoystickButton mrAttackTheesButton4Handling;
025    private boolean Pressed;
026
027    
028    public OI() {
029        
030        // Initialize Joysticks with port numbers
031        xbawks = new Xbox360(1);
032        attack = new Attack3(2);
033        
034        // Initialize all Joystick Buttons
035        targetTrackerButton = xbawks.GetXButton();
036        injectorButton = attack.GetButton(1);
037        specialSweetSpotButton = attack.GetButton(2);
038        mrAttackTheesButton4Handling = attack.GetButton(10);
039        Pressed = false;
040        
041        /**
042         * Map buttons to commands.
043         * whileHeld() executes a command and then calls Interrupted() when
044         * button is released.
045         * 
046         * whenPressed() executes a command once after button is pressed and
047         * released
048         */
049        targetTrackerButton.whenPressed(new AutoAimCommand());
050        injectorButton.whileHeld(new InjectionCommand());
051        specialSweetSpotButton.whileHeld(new SweetSpotCommand());
052        mrAttackTheesButton4Handling.whileHeld(new HandlingCommand());
053    }
054
055    // Getter functions for all Joysticks
056    public Xbox360 getXbawks() {
057        return xbawks;
058    }
059    
060    public Attack3 getAttack3() {
061        return attack;
062    }
063    
064    public boolean getAutoAimState(){
065        return targetTrackerButton.get();
066    }
067}

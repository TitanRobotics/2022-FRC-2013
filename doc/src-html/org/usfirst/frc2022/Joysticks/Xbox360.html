181    /**
182     * Get the angle of the left joystick. It is counter-clockwise, 0 is down.
183     *
184     * @param   reflectCircle   Choose whether to reflect over the Y axis.
185     * @return The angle of the left joystick.
186     */
187    public double GetLeftAngle(boolean reflectCircle) {
188        float yAxis = (float) (-1 * GetLeftY());
189        float xAxis;
190
191        if (reflectCircle) {
192            xAxis = (float) GetLeftX();
193        } else {
194            xAxis = (float) (-1 * GetLeftX());
195        }
196
197        double angle = 0;
198
199        if (xAxis == 0 && yAxis == 0) {
200            angle = 0;
201        } else if (xAxis == 0) {
202            if (yAxis > 0) {
203                angle = 0;
204            } else {
205                angle = Math.PI;        //Legacy: Math.PI
206            }
207        } else if (yAxis == 0) {
208            if (xAxis > 0) {
209                angle = (3 * Math.PI / 2);
210            } else {
211                angle = ((Math.PI) / 2);
212            }
213        } else {
214            angle = MathUtils.atan2(yAxis, xAxis);
215        }
216
217        if (angle >= (2 * Math.PI)) {
218            angle = angle - (2 * Math.PI);
219        }
220
221        if (angle < 0) {
222            angle = angle + (2 * Math.PI);
223        }
224
225        return (angle);
226    }
227
228    /**
229     * Get the angle of the right joystick. It is counter-clockwise, 0 is down.
230     *
231     * @param reflectCircle Choose whether to reflect over the Y axis.
232     * @return The angle of the right joystick.
233     */
234    public double GetRightAngle(boolean reflectCircle) {
235        float yAxis = (float) (-1 * GetRightY());
236        float xAxis;
237
238        if (reflectCircle) {
239            xAxis = (float) GetRightX();
240        } else {
241            xAxis = (float) (-1 * GetRightX());
242        }
243
244        double angle = 0;
245
246        if (xAxis == 0 && yAxis == 0) {
247            angle = 0;
248        } else if (xAxis == 0) {
249            if (yAxis > 0) {
250                angle = 0;
251            } else {
252                angle = Math.PI;
253            }
254        } else if (yAxis == 0) {
255            if (xAxis > 0) {
256                angle = (3 * Math.PI / 2);
257            } else {
258                angle = ((Math.PI) / 2);
259            }
260        } else {
261            angle = MathUtils.atan2(yAxis, xAxis);
262        }
263
264        if (angle >= (2 * Math.PI)) {
265            angle = angle - (2 * Math.PI);
266        }
267
268        if (angle < 0) {
269            angle = angle + (2 * Math.PI);
270        }
271
272        return (angle);
273    }

275    /**
276     * Get the magnitude of the left joystick. Value is between 0 and 1.
277     *
278     * @param
279     * @return The magnitude of the left joystick.
280     */

282        return (Limit(Math.sqrt(MathUtils.pow(GetLeftX(), (double) (2)) + MathUtils.pow(GetLeftY(), (double) (2)))));
283    }

293    }

379    /**
380     * Get the buttons of the controller, to be used by other functions.
381     *
382     * The buttons on the controller follow this mapping 1: A 2: B 3: X 4: Y 5:
383     * Left Bumper 6: Right Bumper 7: Back 8: Start 9: Left Joystick 10: Right
384     * Joystick
385     *
386     * @param button The button to get the value of. Range 1-10.
387     * @return The state of the button.
388     */
389    public boolean GetRawButton(int button) {
390        return (super.getRawButton(button));
391    }

393    /**
394     * Get the raw axes of the controller, to be used by other functions.
395     *
396     * The axis on the controller follow this mapping 1: Left Stick X Axis
397     * -Left:Negative ; Right: Positive 2: Left Stick Y Axis -Up: Negative ;
398     * Down: Positive 3: Triggers -Left: Positive ; Right: Negative 4: Right
399     * Stick X Axis -Left: Negative ; Right: Positive 5: Right Stick Y Axis -Up:
400     * Negative ; Down: Positive 6: Directional Pad (Not recommended, buggy)
401     *
402     * @param axis The axis to get the value of. Range 1-6.
403     * @return The value of the axis from -1 to 1.
404     */
405    public float GetRawAxis(int axis) {
406        return (float) (super.getRawAxis(axis));
407    }

409    /**
410     * Normalize outputs to be between -1 and 1.
411     *
412     * @param num The value to normalize.
413     * @return The normalized value.
414     */
415    public double Limit(double num) {
416        if (num > 1) {
417            num = 1;
418        } else if (num < -1) {
419            num = -1;
420        }
421        return num;
422    }

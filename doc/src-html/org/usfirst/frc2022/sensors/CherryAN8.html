001package org.usfirst.frc2022.sensors;
002
003import edu.wpi.first.wpilibj.AnalogChannel;
004import edu.wpi.first.wpilibj.PIDSource;
005
006import com.sun.squawk.util.MathUtils;
007import java.lang.Math;
008
009/**
010* Functions for interfacing with the Cherry AN8 Magnetic Position sensors.
011* All functions are accessible using the CherryAN8 class
012* 
013* @param channel - The PWM channel on the default module where the sensor is connected (minimum)
014*/
015public class CherryAN8 implements PIDSource{
016        
017        private AnalogChannel sensor;                   //Sensor object used to interact with the hardware of the CRIO
018        private double offset;
019        private double minVoltage;
020        private double maxVoltage;
021        private double voltsPerRadian;
022        
023        /**
024         * Constructor for a Cherry AN8 Magnetic Position Sensor
025         * 
026         * This constructor assumes the minimum voltage possible is .5 volts, and the max voltage possible is 4.5 volts
027         * If that is not the case then call setMinVoltage() and setMaxVoltage() to change those values
028         * 
029         * This constructor also believes that the offset for the sensor is 0, if that is not the case then call the other constructor.
030         * 
031         * @param channel - The PWM channel on the default module where the sensor is connected
032         */
033        public CherryAN8(int channel){
034                this.sensor = new AnalogChannel(channel);
035                this.setOffset(0, true);
036                
037                this.minVoltage = 0.5;
038                this.maxVoltage = 4.5;
039                this.computeVoltsPerRadian();
040                
041        }
042        
043        /**
044         * Constructor for a Cherry AN8 Magnetic Position Sensor
045         * 
046         * This constructor assumes the minimum voltage possible is .5 volts, and the max voltage possible is 4.5 volts
047         * If that is not the case then call setMinVoltage() and setMaxVoltage() to change those values
048         * 
049         * @param channel - The PWM channel on the default module where the sensor is connected
050         * @param offset - The angle the sensor is off from 0, can be anywhere between PI and -PI if  in Radians 180 and -180 if in degrees
051         * @param offsetIsRadians - Boolean value saying if the offset parameter is in radians
052         */
053        public CherryAN8(int channel, double offset, boolean offsetIsRadians){
054                this.sensor = new AnalogChannel(channel);
055                this.setOffset(offset, offsetIsRadians);
056                
057                this.minVoltage = 0.5;
058                this.maxVoltage = 4.5;
059                this.computeVoltsPerRadian();
060        }
061        
062        /**
063         * Constructor for a Cherry AN8 Magnetic Position Sensor
064         * 
065         * This constructor assumes the minimum voltage possible is .5 volts, and the max voltage possible is 4.5 volts
066         * If that is not the case then call setMinVoltage() and setMaxVoltage() to change those values
067         * 
068         * This constructor also believes that the offset for the sensor is 0, if that is not the case then call the other constructor
069         * 
070         * @param slot - The slot in the CRIO where the analog module is placed
071         * @param channel - The PWM channel on the module where the sensor is connected
072         */
073        public CherryAN8(int slot, int channel){
074                this.sensor = new AnalogChannel(slot, channel);
075                this.setOffset(0, true);
076                
077                this.minVoltage = 0.5;
078                this.maxVoltage = 4.5;
079                this.computeVoltsPerRadian();
080        }
081        
082        /**
083         * Constructor for a Cherry AN8 Magnetic Position Sensor
084         * 
085         * This constructor assumes the minimum voltage possible is .5 volts, and the max voltage possible is 4.5 volts
086         * If that is not the case then call setMinVoltage() and setMaxVoltage() to change those values
087         * 
088         * @param slot - The slot in the CRIO where the analog module is placed
089         * @param channel - The PWM channel on the module where the sensor is connected
090         * @param offset - The angle the sensor is off from 0, can be anywhere between PI and -PI if  in Radians 180 and -180 if in degrees
091         * @param offsetIsRadians - Boolean value saying if the offset parameter is in radians
092         */
093        public CherryAN8(int slot, int channel, double offset, boolean offsetIsRadians){
094                this.sensor = new AnalogChannel(slot, channel);
095                this.setOffset(offset, offsetIsRadians);
096                
097                this.minVoltage = 0.5;
098                this.maxVoltage = 4.5;
099                this.computeVoltsPerRadian();
100        }
101        
102        /**
103         * Function setting the offset angle of the sensor
104         * 
105         * @param offset - a value representing the offset of the sensor, can be readians or degrees (set boolean accordingly)
106         * @param isRadians - a boolean value saying whether the inputed value is radians or degrees
107         */
108        public void setOffset(double offset, boolean isRadians){
109                if(isRadians){
110                        this.offset = offset;
111                }
112                else{
113                        this.offset = (offset * (Math.PI/180));
114                }
115        }
116        
117        /**
118         * Function to return the current offset of the sensor
119         * 
120         * @return The current sensor offset in radians
121         */
122        public double getOffset(){
123                return this.offset;
124        }
125        
126        /**
127         * Function to set the maximum voltage of the sensor
128         * This function will call the computeVoltsPerRadian function to adjust for the new voltage.
129         * 
130         * @param maxVoltage - The maximum voltage possible on the sensor, any voltage read higher than this is ignored
131         */
132        public void setMaxVoltage(double maxVoltage){
133                this.maxVoltage = maxVoltage;
134                this.computeVoltsPerRadian();
135        }
136        
137        /**
138         * Function to return the current maximum voltage setting for this sensor
139         * 
140         * @return The current maximum voltage setting for this sensor
141         */
142        public double getMaxVoltage(){
143                return this.maxVoltage;
144        }
145        
146        /**
147         * Function to set the minimum voltage of the sensor
148         * This function will call the computeVoltsPerRadian function to adjust for the new voltage.
149         * 
150         * @param minVoltage - The minimum voltage possible on the sensor, any voltage read lower than this is ignored
151         */
152        public void setMinVoltage(double minVoltage){
153                this.minVoltage = minVoltage;
154                this.computeVoltsPerRadian();
155        }
156        
157        /**
158         * Function to return the current minimum voltage setting for this sensor.
159         * 
160         * @return The current minimum voltage setting for this sensor
161         */
162        public double getMinVoltage(){
163                return this.minVoltage;
164        }
165        
166        /**
167         * Function to compute the voltage per radian for the sensor.
168         * This function should be called anytime a change is made to the minVoltage value or the maxVoltage value
169         * 
170         */
171        public void computeVoltsPerRadian(){
172                double voltageDifference;
173                voltageDifference = this.maxVoltage - this.minVoltage;
174                
175                this.voltsPerRadian = (Math.PI * 2)/voltageDifference;
176        }
177
178        /**
179         * Function to return the current angle of the sensor
180         * 
181         * @return The current angle of the sensor in radians
182         */
183        public double getSensorAngle(){
184                double voltage = this.getSensorVoltage();
185                double angle;
186                
187                if(voltage < this.minVoltage){
188                        voltage = this.minVoltage;
189                }
190                else if(voltage > this.maxVoltage){
191                        voltage = this.maxVoltage;
192                }
193                
194                angle = (voltage - this.minVoltage) * this.voltsPerRadian;
195                angle = angle - this.offset;
196                
197                if(angle == (Math.PI * 2)){
198                        angle = 0;
199                }
200                else if(angle <0){
201                        angle = angle + (Math.PI * 2);
202                }
203                else if(angle > (Math.PI * 2)){
204                        angle = angle - (Math.PI * 2);
205                }
206                
207                return angle;
208        }
209        
210        /**
211         * Function to return the current angle of the sensor
212         * 
213         * @param isRadians - Whether the output should be in radians, Radians if true, Degrees if false
214         * @return The current angle of the sensor in whichever measurement is specified
215         */
216        public double getSensorAngle(boolean isRadians){
217                if(isRadians){
218                        return this.getSensorAngle();
219                }
220                else{
221                        return MathUtils.round(this.getSensorAngle() * (Math.PI/180));
222                }
223        }
224        
225        /**
226         * Function to return the current voltage of the sensor
227         * No adjustments are made to the voltage in this function, this is the raw voltage.
228         * 
229         * @return The raw voltage of the sensor
230         */
231        public double getSensorVoltage(){
232                return this.sensor.getVoltage();
233        }
234        
235        /**
236         * Function to implement PIDSource, giving the result sensor angle to the 
237         * PIDController.
238         * 
239         * @return getSensorAngle()
240         */
241        public double pidGet(){
242                return this.getSensorAngle();
243        }
244}

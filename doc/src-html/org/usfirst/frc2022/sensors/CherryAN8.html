001package org.usfirst.frc2022.sensors;
002
003import edu.wpi.first.wpilibj.AnalogChannel;
004import edu.wpi.first.wpilibj.PIDSource;
005
006import com.sun.squawk.util.MathUtils;
007import java.lang.Math;
008
009/**
010 * Functions for interfacing with the Cherry AN8 Magnetic Position sensors. All
011 * functions are accessible using the CherryAN8 class
012 * 
013* @param channel - The PWM channel on the default module where the sensor is
014 * connected (minimum)
015 */
016public class CherryAN8 implements PIDSource {
017
018    private AnalogChannel sensor;                       //Sensor object used to interact with the hardware of the CRIO
019    private double offset;
020    private double minVoltage;
021    private double maxVoltage;
022    private double voltsPerRadian;
023
024    /**
025     * Constructor for a Cherry AN8 Magnetic Position Sensor
026     *
027     * This constructor assumes the minimum voltage possible is .5 volts, and
028     * the max voltage possible is 4.5 volts If that is not the case then call
029     * setMinVoltage() and setMaxVoltage() to change those values
030     *
031     * This constructor also believes that the offset for the sensor is 0, if
032     * that is not the case then call the other constructor.
033     *
034     * @param channel - The PWM channel on the default module where the sensor
035     * is connected
036     */
037    public CherryAN8(int channel) {
038        this.sensor = new AnalogChannel(channel);
039        this.setOffset(0, true);
040
041        this.minVoltage = 0.5;
042        this.maxVoltage = 4.5;
043        this.computeVoltsPerRadian();
044
045    }
046
047    /**
048     * Constructor for a Cherry AN8 Magnetic Position Sensor
049     *
050     * This constructor assumes the minimum voltage possible is .5 volts, and
051     * the max voltage possible is 4.5 volts If that is not the case then call
052     * setMinVoltage() and setMaxVoltage() to change those values
053     *
054     * @param channel - The PWM channel on the default module where the sensor
055     * is connected
056     * @param offset - The angle the sensor is off from 0, can be anywhere
057     * between PI and -PI if in Radians 180 and -180 if in degrees
058     * @param offsetIsRadians - Boolean value saying if the offset parameter is
059     * in radians
060     */
061    public CherryAN8(int channel, double offset, boolean offsetIsRadians) {
062        this.sensor = new AnalogChannel(channel);
063        this.setOffset(offset, offsetIsRadians);
064
065        this.minVoltage = 0.5;
066        this.maxVoltage = 4.5;
067        this.computeVoltsPerRadian();
068    }
069
070    /**
071     * Constructor for a Cherry AN8 Magnetic Position Sensor
072     *
073     * This constructor assumes the minimum voltage possible is .5 volts, and
074     * the max voltage possible is 4.5 volts If that is not the case then call
075     * setMinVoltage() and setMaxVoltage() to change those values
076     *
077     * This constructor also believes that the offset for the sensor is 0, if
078     * that is not the case then call the other constructor
079     *
080     * @param slot - The slot in the CRIO where the analog module is placed
081     * @param channel - The PWM channel on the module where the sensor is
082     * connected
083     */
084    public CherryAN8(int slot, int channel) {
085        this.sensor = new AnalogChannel(slot, channel);
086        this.setOffset(0, true);
087
088        this.minVoltage = 0.5;
089        this.maxVoltage = 4.5;
090        this.computeVoltsPerRadian();
091    }
092
093    /**
094     * Constructor for a Cherry AN8 Magnetic Position Sensor
095     *
096     * This constructor assumes the minimum voltage possible is .5 volts, and
097     * the max voltage possible is 4.5 volts If that is not the case then call
098     * setMinVoltage() and setMaxVoltage() to change those values
099     *
100     * @param slot - The slot in the CRIO where the analog module is placed
101     * @param channel - The PWM channel on the module where the sensor is
102     * connected
103     * @param offset - The angle the sensor is off from 0, can be anywhere
104     * between PI and -PI if in Radians 180 and -180 if in degrees
105     * @param offsetIsRadians - Boolean value saying if the offset parameter is
106     * in radians
107     */
108    public CherryAN8(int slot, int channel, double offset, boolean offsetIsRadians) {
109        this.sensor = new AnalogChannel(slot, channel);
110        this.setOffset(offset, offsetIsRadians);
111
112        this.minVoltage = 0.5;
113        this.maxVoltage = 4.5;
114        this.computeVoltsPerRadian();
115    }
116
117    /**
118     * Function setting the offset angle of the sensor
119     *
120     * @param offset - a value representing the offset of the sensor, can be
121     * readians or degrees (set boolean accordingly)
122     * @param isRadians - a boolean value saying whether the inputed value is
123     * radians or degrees
124     */
125    public void setOffset(double offset, boolean isRadians) {
126        if (isRadians) {
127            this.offset = offset;
128        } else {
129            this.offset = (offset * (Math.PI / 180));
130        }
131    }
132
133    /**
134     * Function to return the current offset of the sensor
135     *
136     * @return The current sensor offset in radians
137     */
138    public double getOffset() {
139        return this.offset;
140    }
141
142    /**
143     * Function to set the maximum voltage of the sensor This function will call
144     * the computeVoltsPerRadian function to adjust for the new voltage.
145     *
146     * @param maxVoltage - The maximum voltage possible on the sensor, any
147     * voltage read higher than this is ignored
148     */
149    public void setMaxVoltage(double maxVoltage) {
150        this.maxVoltage = maxVoltage;
151        this.computeVoltsPerRadian();
152    }
153
154    /**
155     * Function to return the current maximum voltage setting for this sensor
156     *
157     * @return The current maximum voltage setting for this sensor
158     */
159    public double getMaxVoltage() {
160        return this.maxVoltage;
161    }
162
163    /**
164     * Function to set the minimum voltage of the sensor This function will call
165     * the computeVoltsPerRadian function to adjust for the new voltage.
166     *
167     * @param minVoltage - The minimum voltage possible on the sensor, any
168     * voltage read lower than this is ignored
169     */
170    public void setMinVoltage(double minVoltage) {
171        this.minVoltage = minVoltage;
172        this.computeVoltsPerRadian();
173    }
174
175    /**
176     * Function to return the current minimum voltage setting for this sensor.
177     *
178     * @return The current minimum voltage setting for this sensor
179     */
180    public double getMinVoltage() {
181        return this.minVoltage;
182    }
183
184    /**
185     * Function to compute the voltage per radian for the sensor. This function
186     * should be called anytime a change is made to the minVoltage value or the
187     * maxVoltage value
188     *
189     */
190    public void computeVoltsPerRadian() {
191        double voltageDifference;
192        voltageDifference = this.maxVoltage - this.minVoltage;
193
194        this.voltsPerRadian = (Math.PI * 2) / voltageDifference;
195    }
196
197    /**
198     * Function to return the current angle of the sensor
199     *
200     * @return The current angle of the sensor in radians
201     */
202    public double getSensorAngle() {
203        double voltage = this.getSensorVoltage();
204        double angle;
205
206        if (voltage < this.minVoltage) {
207            voltage = this.minVoltage;
208        } else if (voltage > this.maxVoltage) {
209            voltage = this.maxVoltage;
210        }
211
212        angle = (voltage - this.minVoltage) * this.voltsPerRadian;
213        angle = angle - this.offset;
214
215        if (angle == (Math.PI * 2)) {
216            angle = 0;
217        } else if (angle < 0) {
218            angle = angle + (Math.PI * 2);
219        } else if (angle > (Math.PI * 2)) {
220            angle = angle - (Math.PI * 2);
221        }
222
223        return angle;
224    }
225
226    /**
227     * Function to return the current angle of the sensor
228     *
229     * @param isRadians - Whether the output should be in radians, Radians if
230     * true, Degrees if false
231     * @return The current angle of the sensor in whichever measurement is
232     * specified
233     */
234    public double getSensorAngle(boolean isRadians) {
235        if (isRadians) {
236            return this.getSensorAngle();
237        } else {
238            return MathUtils.round(this.getSensorAngle() * (Math.PI / 180));
239        }
240    }
241
242    /**
243     * Function to return the current voltage of the sensor No adjustments are
244     * made to the voltage in this function, this is the raw voltage.
245     *
246     * @return The raw voltage of the sensor
247     */
248    public double getSensorVoltage() {
249        return this.sensor.getVoltage();
250    }
251
252    /**
253     * Function to implement PIDSource, giving the result sensor angle to the
254     * PIDController.
255     *
256     * @return getSensorAngle()
257     */
258    public double pidGet() {
259        return this.getSensorAngle();
260    }
261}

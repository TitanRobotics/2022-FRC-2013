009public class MaxbotixUltrasonic extends SensorBase {

012    /**
013        * Maxbotix Ultrasonic.
014        * Create it. Good. Now use it.

016    public MaxbotixUltrasonic(int channel){
017        this.channel=new AnalogChannel(channel);
018    }
019    
020    // Just get the voltage.
021        
022        /**
023        * Get the raw voltage from the sensor.
024        * If you're using this, you're doing it wrong.
025        */
026    double GetVoltage() {
027        return channel.getVoltage();
028    }
029    
030    /**
031        * Get the distance!!!
032        * Use this!
033        */
034        double GetDistance(){
035        return GetVoltage()*106.23+0.3793;
036    }
